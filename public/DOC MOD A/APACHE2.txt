    <Location />
        AuthType Basic
        AuthName "Restricted Content"
        AuthUserFile /etc/apache2/.htpasswd

    </Location>

<IfModule ssl_module>
	Listen 443
	Listen 3900
</IfModule>
a2dissite 000-default.conf
a2ensite default-ssl.conf
a2enmod ssl
htpasswd -c /etc/apache2/.htpasswd admin

systemctl restart apache2
