table inet filter {
        chain input {
                type filter hook input priority filter;policy drop;
                ip protocol icmp icmp type echo-request accept; #Autoriser le ping 
                tcp dport 443 accept; #NGINX PROXY
                tcp dport 80 accept; #NGINX PROXY
                udp dport 123 accept; #NTP
                tcp dport 1194 accept; #VPN
                tcp dport 2222 accept; #SSH
                udp dport 53 accept; #DNS
                ct state established,related accept#OBLIGATOIRE
        }
        chain forward {
                type filter hook forward priority filter;policy drop;
                ct state established,related accept#OBLIGATOIRE
                tcp dport 445 accept; #SAMBA
                tcp dport 389 accept; #LDAP
                tcp dport 443 accept; #HTTPS SITE
                tcp dport 3900 accept; #HTTPS SITE
                tcp dport 21 accept; #FTP
                udp dport 53 accept; #DNS
                tcp dport 25 accept; #MAIL-SMTP
                tcp dport 587 accept; #MAIL-SMTP-STARTTLS
                tcp dport 143 accept; #MAIL-IMAP-STARTTLS
                tcp dport 993 accept; #MAIL-IMAP-SSL
                ip saddr 10.1.10.0/24 ip daddr 10.1.20.0/24 accept #Autoriser tout le trafic de INT vers DMZ
                ip saddr 10.1.20.0/24 ip daddr 10.1.10.0/24 accept #Autoriser tout le trafic de DMZ vers INT
        }
        chain output {
                type filter hook output priority filter;policy accept;
        }
}
