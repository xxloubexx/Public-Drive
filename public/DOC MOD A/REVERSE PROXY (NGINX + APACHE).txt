NGINX

apt install nginx
cp /etc/nginx/sites-enabled/default /etc/nginx/sites-enabled/home
/etc/nginx/sites-enabled/home
listen 80;
server_name home.patiobrancion.com;
location / {
	proxy_pass http://10.1.20.1;
}

cp /etc/nginx/sites-enabled/home /etc/nginx/sites-enabled/room
listen 443 ssl;
server_name room.patiobrancion.com;
ssl_certificate /home/worldskills/room.crt;
ssl_certificate_key /home/worldskills/room.key;
location / {
	proxy_pass http://10.1.20.1; #le site est de base en http avec le reverse proxy on le transforme en https
}

systemctl restart nginx
