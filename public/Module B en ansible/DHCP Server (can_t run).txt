- name: Create DHCP Server
  hosts: addomain
  tasks:
    - name: Installation DHCP Service
      ansible.windows.win_feature:
        name: "DHCP"
    - name: ADD Scope DHCP
      win_shell: |
        Add-DhcpServerV4Scope -Name "AnsibleScope" -StartRange 10.10.0.50 -EndRange 10.10.0.100 -SubnetMask 255.255.255.0 -State Active
    - name: Add Options Scope
      win_shell: |
        Set-DhcpServerv4OptionValue -ScopeId 10.10.0.0 -Router 10.10.0.1 -DnsServer 10.10.0.10
